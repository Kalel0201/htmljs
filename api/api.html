Data

1. Get and set

html.data give you ability to observe any kind of object - from basic types like String Number Boolean Array and Function.

You want to observe a string. 

var helloWorld = html.data('Hello world!');
Get the value of helloWorld variable. 
//get value
helloWorld(); // "Hello world!"
Set the value of helloWorld variable. 
//set value
hellowWorld('Hello HTML Engine');
Other type to observe. 
var a = html.data(1),
    b = html.data(2),
    computed = html.data(function() {
        return a() + b();
    });
    //computed() => 3

var array = html.data([1,2,3,4,5]);

2. changeAfter

-method

Register dependency of a computed value.

ParamTypeDetails
arguments html.data

Register as many dependencies as you want.

 
var firstName = html.data('Henry'),
    lastName = html.data('Ford'),
    fullName = html.data(function() {
        return firstName() + ' ' + lastName();
    });

//fullName() ==> "Henry Ford"

3. delay

-property

Set delay time to notify change to UI and its dependencies. 

var test = html.data('Someone');
    test.delay = 100;
//now every time an input change test, all UI and its dependencies will be notified after 100 milliseconds.

4. dependencies

-property

Get all dependencies from html.data. 

var a = html.data(4),
b = html.data(2);
c = html.data(function() {
    return a*b;
}).changeAfter(a, b);

//c.dependencies ==> [a,b]

5. isComputed

Check the data is kind of function (aka computed).

var firstName = html.data('Henry'),
    lastName = html.data('Ford'),
    fullName = html.data(function() {
        return firstName() + ' ' + lastName();
    });
//lastName.isComputed() ==> false
//fullName.isComputed() ==> true

6. lazyInput

-property

Set this property for indicate when the observed value change.
If true then the value only change after change event.
If false then the value will change immediately when key stroke.

7. Refresh

-method

Use refresh method for update data manually, especially when working with complex computed values.

hellowWorld.refresh();

8. setValidationResult

-method

Use to set validation result.

ParamTypeDetails
isValid Boolean

True if the data is valid.
message String

Optional. Message to show when data is not valid.

 
// custom validation rule
html.data.validation.beginWith = function(char, message) {
    this.validate(function(newValue, oldvalue) {
        if(newValue.charAt(0).toLowerCase() !== char.toLowerCase()) {
            this.setValidationResult(false, message);
        } else {
            this.setValidationResult(true);
        }
    });
    return this;
};
// custom validation rule usage
var name = html.data('Someone').beginWith('a');

9. silentSet

-method

Use to set value without notifying UI and dependencies.

ParamTypeDetails
value Object

The new value.

 
var name = html.data('Someone');
name.silent('Another name');

10. subscribe

-method

Subscribe a function to be run when the value changed.

ParamTypeDetails
arg Function<object, object>

Function to run when the value of observer changed.

 
//set value
hellowWorld.subscribe(function(value, lastValue) {
    console.log(value, lastValue);
});

11. targets

-property

All functions that subscribed to this observer.

12. unsubscribe

-method

Unsubscribe a function from observer. This method will remove a function in html-data-targets

var firstName = html.data('Henry'),
sub = function(val) {
    console.log(val);
};
firstName.subscribe(sub);
// firstName.targets.length ==> 1
firstName.unsubscribe(sub);
// firstName.targets.length ==> 0

13. validate

-property

Use to create custom validation rules.

ParamTypeDetails
arguments Function<oldValue,newValue>

The callback function will have 2 parameters: oldValue and newValue.
See the example in setValidationResult

14. validators

-property

All validators of the observer.

15. validationResults

-property

All validationResults of the observer.

Data <Array>

You can use these methods as following when you need to observe an array.

1. add

-method

Add an object to the list, and the UI will be automatically updated.

ParamTypeDetails
object Object

New object to add.
index Number

(Optional - default 0) Index for new item.

 
var list = html.data([1,2,3,4,5,6]);
list.add(7);
list.add(0, 0);

2. filter

-method -no return

Filter a list and update the searching result.

ParamTypeDetails
searchString String

Search string.

 
var list = html.data(['Anderson', 'Peter', 'Robin']);
list.filter('an'); // ==> ["Anderson"]
list.filter('er'); // ==> ["Anderson", "Peter"]

3. getFilterResult

-method -return Array filtered list

Return the filter result.

var list = html.data(['Anderson', 'Peter', 'Robin']);
list.filter('an'); list.getFilterResult(); // ==> ["Anderson"]
list.filter('er'); list.getFilterResult(); // ==> ["Anderson", "Peter"]

4. move

-method -no return

Move an item in the list to a new position, also update UI.

ParamTypeDetails
position Number

Position of an item.
newPosition Number

New position to move to.

 
var list = html.data(['Anderson', 'Peter', 'Robin']);
list.move(0, 2);
//list() ==> ['Peter', 'Robin', 'Anderson']

5. orderBy

-method -return this Function

Order a list by some criteria. Support multiple level order.

ParamTypeDetails
arguments String Function Array

You can pass one of these types to order the list.

 
var people = html.data([
    {name: 'EEE', age: 20},
    {name: 'ABC', age: 50},
    {name: 'BBB', age: 25}
]);
people.orderBy('name'); // ==> [{name: 'ABC', age: 50}, {name: 'BBB', age: 25}, {name: 'EEE', age: 20}]
people.orderBy('age'); // ==> [{name: 'EEE', age: 20}, name: 'BBB', age: 25}, {name: 'ABC', age: 50}]
people.orderBy({field: 'name', isAsc: false}); // ==> [{name: 'EEE', age: 20}, name: 'BBB', age: 25}, {name: 'ABC', age: 50}]
people.orderBy({field: 'age', isAsc: false}); // ==> [{name: 'ABC', age: 50}, {name: 'BBB', age: 25}, {name: 'EEE', age: 20}]
people.orderBy([{field: 'age', isAsc: false}, {field: 'age', isAsc: true}]); // ==> [{name: 'ABC', age: 50}, {name: 'BBB', age: 25}, {name: 'EEE', age: 20}]

var orderFn = function(x) {return x.name; };
orderFn.isAsc = false;
people.orderBy(orderFn); // equal to people.orderBy({field: 'name', isAsc: false});
